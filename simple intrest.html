/* Algorithm
01. Start
02. Accept Principle (p), Rate of interest (r), Time period(t)
03. Display Simple Interest after evaluating p * t * r / 100
04. Stop */
# include <stdio.h>
int main() {
int p, t, r;
printf("Principle = " );
scanf("%d", &p);
printf("Rate of Interest = " );
scanf("%d", &r);
printf("Time period = " );
scanf("%d", &t);
printf("Simple Interest = %d \n", (p * t * r / 100) );
return 0;
}
